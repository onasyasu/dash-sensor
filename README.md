# usage
```
sudo vi /boot/config.txt
    dtparam=spi=on
sudo reboot    

sudo apt-get install git
sudo apt-get install python-dev

mkdir git
git clone git://github.com/doceme/py-spidev
cd py-spidev
sudo python setup.py install

cd ../
git clone https://github.com/tk3fftk/dash-sensor.git 
cd dash-sensor
sudo python main.py
```
